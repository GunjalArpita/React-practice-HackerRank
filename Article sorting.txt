App.js


import "h8k-components";

import Articles from "./components/Articles";
import {useState} from "react"
import "./App.css";


function App({ articles }) {
  const initialCopy = [...articles].sort((a,b) => b.upvotes - a.upvotes) 
const[sortedArticles,setSortedArticles] =useState(initialCopy)


  const handleMostUpvoted = () => {
    // Logic for most upvoted articles
    const copied =[...sortedArticles];
    copied.sort((a,b)=> b.upvotes-a.upvotes)
    setSortedArticles(copied)
    

  };

  const handleMostRecent = () => {
    // Logic for most recent articles
    const copied =[...sortedArticles];
    copied.sort((a,b) => new Date(b.date) - new Date(a.date))
    setSortedArticles(copied)


  };
  return (
    <>
      <h8k-navbar header="Sorting Articles"></h8k-navbar>
      <div className="App">
        <div className="layout-row align-items-center justify-content-center my-20 navigation">
          <label className="form-hint mb-0 text-uppercase font-weight-light">
            Sort By
          </label>
          <button
            data-testid="most-upvoted-link"
            className="small"
            onClick={handleMostUpvoted}
          >
            Most Upvoted
          </button>
          <button
            data-testid="most-recent-link"
            className="small"
            onClick={handleMostRecent}
          >
            Most Recent
          </button>
        </div>
        <Articles articles={sortedArticles} />
      </div>
    </>
  );
}

export default App;




Article.js


import React from "react";

function Articles({ articles = [] }) {
  return (
    <div className="card w-50 mx-auto">
      <table>
        <thead>
         
              <tr>
            <th >Title </th>
            <th >Upvotes </th>
            <th >Date</th>
          </tr>

        </thead>
        <tbody>
          {
            articles.map((article,index) => {
              return(
            <tr data-testid="article" key={index}>
            <td data-testid="article-title">{article.title}</td>
            <td data-testid="article-upvotes">{article.upvotes}</td>
            <td data-testid="article-date">{article.date}</td>
          </tr>
              )
            })
          }
          
        </tbody>
      </table>
    </div>
  );
}

export default Articles;
